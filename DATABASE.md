# Database Schema

## Conventions

- All tables use `UUID` primary keys (generated by `gen_random_uuid()`)
- All tables include `created_at`, `updated_at` (auto-managed), and `created_by` (nullable for system-generated rows)
- Soft delete via `deleted_at TIMESTAMPTZ` — hard delete only for GDPR erasure
- All column names are `snake_case`
- All tables have Row Level Security enabled — no exceptions
- Foreign keys use `_id` suffix (e.g., `dispute_id`, `party_id`)
- Enums are PostgreSQL enums, not strings — validates at the database level
- Timestamps are always `TIMESTAMPTZ` (with timezone), stored as UTC

## Core Tables

### `profiles`

Extended user profile linked to Supabase Auth.

```sql
CREATE TABLE profiles (
  id UUID PRIMARY KEY REFERENCES auth.users(id) ON DELETE CASCADE,
  display_name TEXT NOT NULL,
  business_name TEXT,
  business_type TEXT, -- e.g., 'sole_trader', 'limited_company', 'partnership'
  email TEXT NOT NULL,
  phone TEXT,
  notification_preferences JSONB DEFAULT '{"email": true, "in_app": true}'::jsonb,
  onboarding_completed BOOLEAN DEFAULT FALSE,
  created_at TIMESTAMPTZ DEFAULT NOW(),
  updated_at TIMESTAMPTZ DEFAULT NOW(),
  deleted_at TIMESTAMPTZ
);

-- RLS: users can only read/write their own profile
ALTER TABLE profiles ENABLE ROW LEVEL SECURITY;
CREATE POLICY "Users manage own profile"
  ON profiles FOR ALL
  USING (auth.uid() = id)
  WITH CHECK (auth.uid() = id);
```

### `disputes`

Core dispute record. Created when one party files against another.

```sql
CREATE TYPE dispute_status AS ENUM (
  'draft',
  'filed',
  'awaiting_response',
  'in_progress',
  'resolved',
  'escalated',
  'withdrawn',
  'expired'
);

CREATE TYPE dispute_type AS ENUM (
  'payment',
  'deliverables',
  'service_quality',
  'contract_interpretation',
  'other'
);

CREATE TABLE disputes (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  reference_number TEXT UNIQUE NOT NULL, -- Human-readable: KST-2026-XXXXX
  initiating_party_id UUID NOT NULL REFERENCES profiles(id),
  responding_party_id UUID, -- NULL until responding party is identified/registered
  responding_party_email TEXT NOT NULL, -- Used for invitation if not yet registered
  dispute_type dispute_type NOT NULL,
  status dispute_status NOT NULL DEFAULT 'draft',
  subject TEXT NOT NULL, -- Brief description
  amount_disputed DECIMAL(12,2), -- In GBP, nullable (not all disputes are monetary)
  currency TEXT DEFAULT 'GBP' CHECK (currency = 'GBP'), -- Phase 1: GBP only
  response_deadline TIMESTAMPTZ, -- When the responding party must respond by
  resolution_summary TEXT, -- Filled when resolved
  resolved_at TIMESTAMPTZ,
  escalated_at TIMESTAMPTZ,
  created_at TIMESTAMPTZ DEFAULT NOW(),
  updated_at TIMESTAMPTZ DEFAULT NOW(),
  created_by UUID REFERENCES profiles(id),
  deleted_at TIMESTAMPTZ
);

-- RLS: only named parties can see their disputes
ALTER TABLE disputes ENABLE ROW LEVEL SECURITY;
CREATE POLICY "Parties see own disputes"
  ON disputes FOR SELECT
  USING (
    auth.uid() = initiating_party_id
    OR auth.uid() = responding_party_id
  );

CREATE POLICY "Initiator creates disputes"
  ON disputes FOR INSERT
  WITH CHECK (auth.uid() = initiating_party_id);

CREATE POLICY "Parties update own disputes"
  ON disputes FOR UPDATE
  USING (
    auth.uid() = initiating_party_id
    OR auth.uid() = responding_party_id
  );
```

### `dispute_submissions`

Individual submissions within a dispute. Immutable once created.

```sql
CREATE TYPE submission_type AS ENUM (
  'initial_claim',
  'response',
  'reply',
  'evidence_summary',
  'proposal',
  'acceptance',
  'rejection',
  'withdrawal'
);

CREATE TABLE dispute_submissions (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  dispute_id UUID NOT NULL REFERENCES disputes(id),
  submitted_by UUID NOT NULL REFERENCES profiles(id),
  submission_type submission_type NOT NULL,
  content TEXT NOT NULL,
  content_hash TEXT NOT NULL, -- SHA-256 of content at creation, for tamper detection
  metadata JSONB DEFAULT '{}'::jsonb,
  created_at TIMESTAMPTZ DEFAULT NOW(),
  -- No updated_at: submissions are immutable
  -- No deleted_at: submissions cannot be deleted (audit trail)
  
  -- Ensure submitter is a party to the dispute
  CONSTRAINT submitter_is_party CHECK (TRUE) -- Enforced via RLS
);

ALTER TABLE dispute_submissions ENABLE ROW LEVEL SECURITY;
CREATE POLICY "Parties see own dispute submissions"
  ON dispute_submissions FOR SELECT
  USING (
    submitted_by = auth.uid()
    OR dispute_id IN (
      SELECT id FROM disputes
      WHERE initiating_party_id = auth.uid()
         OR responding_party_id = auth.uid()
    )
  );

CREATE POLICY "Parties create submissions"
  ON dispute_submissions FOR INSERT
  WITH CHECK (
    auth.uid() = submitted_by
    AND dispute_id IN (
      SELECT id FROM disputes
      WHERE initiating_party_id = auth.uid()
         OR responding_party_id = auth.uid()
    )
  );

-- No UPDATE or DELETE policies: submissions are immutable
```

### `evidence_files`

Files attached to disputes as evidence.

```sql
CREATE TABLE evidence_files (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  dispute_id UUID NOT NULL REFERENCES disputes(id),
  submission_id UUID REFERENCES dispute_submissions(id),
  uploaded_by UUID NOT NULL REFERENCES profiles(id),
  file_name TEXT NOT NULL,
  file_type TEXT NOT NULL, -- MIME type
  file_size_bytes BIGINT NOT NULL,
  storage_path TEXT NOT NULL, -- Path in Supabase Storage
  scan_status TEXT DEFAULT 'pending' CHECK (scan_status IN ('pending', 'clean', 'flagged')),
  description TEXT,
  created_at TIMESTAMPTZ DEFAULT NOW()
  -- No updated_at or deleted_at: evidence files are immutable
);

ALTER TABLE evidence_files ENABLE ROW LEVEL SECURITY;
CREATE POLICY "Parties see dispute evidence"
  ON evidence_files FOR SELECT
  USING (
    dispute_id IN (
      SELECT id FROM disputes
      WHERE initiating_party_id = auth.uid()
         OR responding_party_id = auth.uid()
    )
  );

CREATE POLICY "Parties upload evidence"
  ON evidence_files FOR INSERT
  WITH CHECK (
    auth.uid() = uploaded_by
    AND dispute_id IN (
      SELECT id FROM disputes
      WHERE initiating_party_id = auth.uid()
         OR responding_party_id = auth.uid()
    )
  );
```

### `saved_documents`

Documents generated by authenticated users and saved to their account.

```sql
CREATE TYPE document_type AS ENUM (
  'contract',
  'terms_and_conditions',
  'handshake',
  'notice',
  'milestone_tracker',
  'late_payment_letter'
);

CREATE TABLE saved_documents (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id UUID NOT NULL REFERENCES profiles(id),
  document_type document_type NOT NULL,
  title TEXT NOT NULL,
  configuration JSONB NOT NULL, -- The inputs used to generate (for re-generation)
  clause_versions JSONB NOT NULL, -- Snapshot of which clause versions were used
  includes_dispute_clause BOOLEAN NOT NULL DEFAULT TRUE,
  storage_path TEXT, -- Path in Supabase Storage if exported to file
  created_at TIMESTAMPTZ DEFAULT NOW(),
  updated_at TIMESTAMPTZ DEFAULT NOW(),
  deleted_at TIMESTAMPTZ
);

ALTER TABLE saved_documents ENABLE ROW LEVEL SECURITY;
CREATE POLICY "Users manage own documents"
  ON saved_documents FOR ALL
  USING (auth.uid() = user_id)
  WITH CHECK (auth.uid() = user_id);
```

### `audit_log`

Immutable audit trail for all significant actions.

```sql
CREATE TABLE audit_log (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  actor_id UUID REFERENCES profiles(id), -- NULL for system actions
  action TEXT NOT NULL, -- e.g., 'dispute.filed', 'submission.created', 'document.generated'
  resource_type TEXT NOT NULL, -- e.g., 'dispute', 'submission', 'document'
  resource_id UUID NOT NULL,
  metadata JSONB DEFAULT '{}'::jsonb, -- Additional context
  ip_address INET,
  user_agent TEXT,
  created_at TIMESTAMPTZ DEFAULT NOW()
  -- No update or delete: audit logs are immutable
);

ALTER TABLE audit_log ENABLE ROW LEVEL SECURITY;
-- Users can see audit entries for their own actions and their own resources
CREATE POLICY "Users see own audit trail"
  ON audit_log FOR SELECT
  USING (actor_id = auth.uid());

-- Only the application (service role) inserts audit entries
-- No user-facing INSERT policy
```

### `notifications`

User notification queue.

```sql
CREATE TABLE notifications (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id UUID NOT NULL REFERENCES profiles(id),
  dispute_id UUID REFERENCES disputes(id),
  type TEXT NOT NULL, -- e.g., 'dispute_filed', 'new_submission', 'deadline_approaching'
  title TEXT NOT NULL,
  body TEXT NOT NULL,
  read BOOLEAN DEFAULT FALSE,
  email_sent BOOLEAN DEFAULT FALSE,
  email_sent_at TIMESTAMPTZ,
  created_at TIMESTAMPTZ DEFAULT NOW()
);

ALTER TABLE notifications ENABLE ROW LEVEL SECURITY;
CREATE POLICY "Users see own notifications"
  ON notifications FOR SELECT
  USING (auth.uid() = user_id);

CREATE POLICY "Users update own notifications"
  ON notifications FOR UPDATE
  USING (auth.uid() = user_id)
  WITH CHECK (auth.uid() = user_id);
```

## Indexes

```sql
-- Disputes: lookups by party
CREATE INDEX idx_disputes_initiating_party ON disputes(initiating_party_id) WHERE deleted_at IS NULL;
CREATE INDEX idx_disputes_responding_party ON disputes(responding_party_id) WHERE deleted_at IS NULL;
CREATE INDEX idx_disputes_status ON disputes(status) WHERE deleted_at IS NULL;
CREATE INDEX idx_disputes_reference ON disputes(reference_number);

-- Submissions: lookups by dispute
CREATE INDEX idx_submissions_dispute ON dispute_submissions(dispute_id);
CREATE INDEX idx_submissions_author ON dispute_submissions(submitted_by);

-- Evidence: lookups by dispute
CREATE INDEX idx_evidence_dispute ON evidence_files(dispute_id);

-- Documents: lookups by user
CREATE INDEX idx_documents_user ON saved_documents(user_id) WHERE deleted_at IS NULL;

-- Notifications: unread for user
CREATE INDEX idx_notifications_user_unread ON notifications(user_id) WHERE read = FALSE;

-- Audit log: lookups by resource
CREATE INDEX idx_audit_resource ON audit_log(resource_type, resource_id);
CREATE INDEX idx_audit_actor ON audit_log(actor_id);
```

## Migration Discipline

1. Migrations are forward-only. Never edit a migration that has been applied.
2. Every migration has a descriptive name: `20260401_create_disputes_table.sql`
3. Destructive migrations (dropping columns, changing types) require a two-step process:
   - Step 1: Deploy code that handles both old and new schema
   - Step 2: Apply the migration
   - Step 3: Deploy code that only handles the new schema
4. All migrations are tested against a copy of production data before applying to production.
5. Migrations are applied through Supabase CLI, not through the dashboard.
